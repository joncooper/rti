  {

    y: r * sin(phi) * sin(theta)
    z: r * cos(theta)
  }

Convert cartesian coordinates to spherical coordinates phi measures rotation around the xy plane theta measures rotation along the yz plane

 
cartesianToSpherical = (x, y, z) ->
  {
    r:     sqrt(x*x + y*y + z*z)
    theta: acos(z)
    phi:   atan2(y, x)
  }
