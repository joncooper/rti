Lines end with either LF (0x0a) or CRLF (0x0d0a). Note that this will scan until EOF looking for an end-of-line.

 
  readLine: ->
    if @pos >= @dataView.byteLength
      return null

    start = @pos
    end   = -1

    while (@pos < @dataView.byteLength) && (end < start)
      if @dataView.getUint8(@pos) is 0x0a
        if (@pos > 0) && (@dataView.getUint8(@pos-1) is 0x0d)
          end = @pos - 1
        else
          end = @pos
        @pos = @pos + 1
        return @stringFromUint8Slice(start, end)
      else
        @pos = @pos + 1
    return null
